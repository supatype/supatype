# @supatype/client

Typed HTTP client for [Supatype](https://github.com/supatype/supatype). Wraps REST, auth, storage, realtime, and edge functions with full TypeScript inference from your generated database types.

## Installation

```bash
npm i @supatype/client
```

## Setup

```ts
import { createClient } from "@supatype/client"
import type { Database } from "./types/database" // generated by supatype generate

export const supatype = createClient<Database>({
  url: process.env.SUPATYPE_URL!,
  anonKey: process.env.SUPATYPE_ANON_KEY!,
})
```

## Querying data

```ts
// Select with filters
const { data: posts, error } = await supatype
  .from("posts")
  .select()
  .eq("status", "published")
  .order("created_at", { ascending: false })
  .limit(10)

// Select specific columns
const { data } = await supatype
  .from("posts")
  .select("id, title, author_id")
  .eq("id", postId)
  .single()

// Insert
const { data: newPost, error } = await supatype
  .from("posts")
  .insert({ title: "Hello", body: "…", author_id: userId })

// Update
const { error } = await supatype
  .from("posts")
  .update({ title: "Updated" })
  .eq("id", postId)

// Delete
const { error } = await supatype
  .from("posts")
  .delete()
  .eq("id", postId)

// Upsert
const { data } = await supatype
  .from("posts")
  .upsert({ id: postId, title: "Hello" })
```

## Auth

```ts
const { data, error } = await supatype.auth.signUp({ email, password })
const { data, error } = await supatype.auth.signIn({ email, password })
await supatype.auth.signOut()

// Listen for auth state changes
supatype.auth.onAuthStateChange((event, session) => {
  console.log(event, session?.user)
})

// Get the current session
const session = await supatype.auth.getSession()
const user = await supatype.auth.getUser()
```

## Storage

```ts
// Upload a file
const { data, error } = await supatype.storage
  .from("avatars")
  .upload(`${userId}/avatar.jpg`, file)

// Get a public URL
const { data } = supatype.storage
  .from("avatars")
  .getPublicUrl(`${userId}/avatar.jpg`)

// Download
const { data } = await supatype.storage
  .from("avatars")
  .download(`${userId}/avatar.jpg`)

// List files
const { data: files } = await supatype.storage
  .from("avatars")
  .list(userId)

// Delete
const { error } = await supatype.storage
  .from("avatars")
  .remove([`${userId}/avatar.jpg`])
```

## Realtime

```ts
const channel = supatype.realtime.channel("posts")

channel
  .on("INSERT", (payload) => console.log("new post:", payload.new))
  .on("UPDATE", (payload) => console.log("updated:", payload.new))
  .on("DELETE", (payload) => console.log("deleted:", payload.old))
  .subscribe()

// Presence
channel.track({ userId, status: "online" })

// Unsubscribe
channel.unsubscribe()
```

## Edge functions

```ts
const { data, error } = await supatype.functions.invoke("send-email", {
  body: { to: "user@example.com", subject: "Welcome" },
})
```

## Postgres functions (RPC)

```ts
const { data, error } = await supatype.rpc("calculate_shipping", {
  weight: 2.5,
  country: "GB",
})
```

## GraphQL

```ts
const { data, error } = await supatype.graphql(`
  query Posts {
    postsCollection {
      edges { node { id title } }
    }
  }
`)
```

## Singleton globals

```ts
// Fetch a singleton global model (e.g. SiteSettings)
const { data, error } = await supatype.global<SiteSettings>("site_settings").get()
await supatype.global<SiteSettings>("site_settings").update({ siteName: "My App" })
```

## Configuration

```ts
createClient<Database>({
  url: "https://…",
  anonKey: "…",

  // Optional service role key (admin operations, bypasses RLS)
  serviceRoleKey: process.env.SUPATYPE_SERVICE_ROLE_KEY,

  // Retry config
  retry: { attempts: 3, baseDelay: 500 },

  // Request timeout (ms)
  timeout: 10_000,

  // Preload an existing session (SSR)
  initialSession: { access_token: "…", refresh_token: "…" },

  // Auth storage options
  auth: {
    persistSession: true,
    storageKey: "myapp-auth",
    cookiePrefix: "myapp",
  },
})
```
